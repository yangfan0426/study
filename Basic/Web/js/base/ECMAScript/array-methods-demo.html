数组方法分类

创建数组


// 数组字面量
const arr1 = [1, 2, 3];

// Array构造函数
const arr2 = new Array(1, 2, 3);
const arr3 = new Array(3); // 创建长度为3的空数组

// Array.of()
const arr4 = Array.of(1, 2, 3);

// Array.from()
const arr5 = Array.from('hello');
const arr6 = Array.from([1, 2, 3], x => x * 2);

添加/删除元素


const arr = ['a', 'b', 'c'];

// 末尾操作
arr.push('d');  // 添加到末尾
arr.pop();      // 删除末尾元素

// 开头操作
arr.unshift('x'); // 添加到开头
arr.shift();      // 删除开头元素

// 任意位置操作
arr.splice(1, 1);        // 从索引1开始删除1个元素
arr.splice(1, 0, 'y');   // 在索引1处插入'y'
arr.splice(1, 1, 'z');   // 替换索引1处的元素为'z'

数组遍历


const arr = [1, 2, 3, 4, 5];

// 基本遍历
arr.forEach((item, index) => console.log(index, item));

// 映射
const doubled = arr.map(x => x * 2);

// 过滤
const evens = arr.filter(x => x % 2 === 0);

// 归约
const sum = arr.reduce((acc, curr) => acc + curr, 0);

// 查找
const found = arr.find(x => x > 3);
const foundIndex = arr.findIndex(x => x > 3);
